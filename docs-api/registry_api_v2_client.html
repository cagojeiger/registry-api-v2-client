104    """tar 파일이 유효한 Docker 이미지 tar 파일인지 검증합니다.
105
106    Args:
107        tar_path: 검증할 tar 파일 경로
108            - Path 객체: Path("/Users/user/images/app.tar")
109            - 상대경로도 지원: Path("./docker-images/nginx.tar")
110
111    Returns:
112        bool: 유효한 Docker 이미지 tar 파일인 경우 True, 그렇지 않으면 False
113
114    Raises:
115        ValidationError: tar 파일이 손상되었거나 잘못된 형식인 경우
116
117    Examples:
118        # tar 파일 검증
119        from pathlib import Path
120
121        is_valid = validate_docker_tar(Path("nginx.tar"))
122        if is_valid:
123            print("유효한 Docker 이미지 tar 파일입니다")
124        else:
125            print("유효하지 않은 tar 파일입니다")
126    """
127    try:
128        if not is_path_exists(tar_path):
129            raise ValidationError(f"Tar file does not exist: {tar_path}")
130
131        if not is_valid_tarfile(tar_path):
132            return False
133
134        with tarfile.open(tar_path, "r") as tar:
135            tar_members = get_tar_members(tar)
136            required_files = ["manifest.json"]
137
138            if not has_required_files(tar_members, required_files):
139                return False
140
141            manifest_content = extract_manifest_content(tar)
142            if manifest_content is None:
143                return False
144
145            manifest_data = parse_manifest_json(manifest_content)
146            if manifest_data is None:
147                return False
148
149            return validate_all_manifest_entries(manifest_data, tar_members)
150
151    except (tarfile.TarError, OSError) as e:
152        raise ValidationError(f"Error reading tar file: {e}") from e

186    """Docker tar 파일을 검사하여 상세한 이미지 정보를 반환합니다.

188    Args:
189        tar_path: Docker tar 파일 경로

192
193    Returns:
194        ImageInspect: 완전한 이미지 정보 객체 (아키텍처, OS, 레이어, 환경변수 등)
195
196    Raises:
197        ValidationError: tar 파일이 유효하지 않은 경우
198        TarReadError: tar 파일을 읽을 수 없는 경우
199
200    Examples:
201        # Docker tar 파일 상세 검사

209        print(f"환경변수: {inspect_result.config.env}")
210    """
211    if not validate_docker_tar(tar_path):
212        raise ValidationError(f"Invalid Docker tar file: {tar_path}")
213
214    try:
215        with tarfile.open(tar_path, "r") as tar:
216            # Extract manifest
217            manifest_data = extract_json_file(tar, "manifest.json")
218            if not validate_manifest_data(manifest_data):
219                raise TarReadError("Invalid manifest.json")
220
221            # Type guard ensures manifest_data is a list at this point
222            assert isinstance(manifest_data, list)
223            manifest = manifest_data[0]  # Use first image
224
225            # Extract config
226            config_path = manifest["Config"]
227            config_data = extract_json_file(tar, config_path)
228            if not validate_config_data(config_data):
229                raise TarReadError(f"Cannot read config file: {config_path}")
230
231            # Parse layers from manifest
232            layer_paths = manifest.get("Layers", [])
233            layer_sources = manifest.get("LayerSources", {})
234
235            # Build layer info
236            layers, total_size = build_layers_info(tar, layer_paths, layer_sources)
237
238            # Type guard ensures config_data is a dict at this point
239            assert isinstance(config_data, dict)
240            image_config = parse_image_config(config_data)
241
242            # Calculate config digest
243            config_digest = extract_config_digest(config_path)
244
245            # Build final result
246            return build_image_inspect(
247                manifest, image_config, layers, total_size, config_digest
248            )
249
250    except (tarfile.TarError, json.JSONDecodeError, KeyError) as e:
251        raise TarReadError(f"Failed to inspect tar file: {e}") from e

def parse_repository_tag(repo_tag: str) -> tuple[str, str]: 
150def parse_repository_tag(repo_tag: str) -> tuple[str, str]:
151    """저장소:태그 문자열을 저장소와 태그 구성요소로 파싱합니다.
152
153    Args:
154        repo_tag: 저장소 태그 문자열
155            - 예: "nginx:alpine", "localhost:5000/myapp:latest"
156            - 레지스트리 포함: "registry.io/company/app:v1.0"
157
158    Returns:
159        tuple[str, str]: (저장소, 태그) 튜플
160
161    Examples:
162        # 기본 이미지 태그 파싱
163        repo, tag = parse_repository_tag("nginx:alpine")
164        # 결과: ("nginx", "alpine")
165
166        # 레지스트리 포함 태그 파싱
167        repo, tag = parse_repository_tag("localhost:5000/myapp:latest")
168        # 결과: ("localhost:5000/myapp", "latest")
169
170        # 태그 없는 경우 (기본값 사용)
171        repo, tag = parse_repository_tag("myapp")
172        # 결과: ("myapp", "latest")
173    """
174    if ":" in repo_tag:
175        # Split only on the last ':' to handle registry URLs like localhost:5000/repo:tag
176        parts = repo_tag.rsplit(":", 1)
177        if len(parts) == 2 and parts[1]:  # Ensure tag is not empty
178            return parts[0], parts[1]
179        else:
180            # Empty tag after colon (e.g., "app:")
181            return parts[0], "latest"
182
183    # No tag specified, use default
184    return repo_tag, "latest"
